-- INITIALIZE GIT PROJECT
	init git

-- REPO STATUS
	git status

-- ADD FILES TO GIT
	git add .
	
-- COMMIT ADDED FILES AND CHANGES
	git commit -m "initial commit"

-- VIEW ALL COMMIT LOGS
	git log
	git log -n 5 				# shows only the last 5 updates
	git log --since=2020-01-01	# shows logs only since the date
	git log --author="eduard"   # searchs for author (does not need to be full author name)
	git log --grep="Init" 		# searches the commit section with grep (is case sensitive)
	
	complex example
		git log --since=2019-03-01 --until=2019-03-31 --author="Karen" --grep="refactor"

-- VIEW DIFFERENCE BETWEEN FILES
	git diff 			# shows diff only for files in working area
	git diff --staged	# shows diff only on staged area
	
	git diff --color-words  # show in color only words that changed
	
-- remove file from git
	git rm file_to_delete2.txt
	git commit -m "delete file 2"
	
-- moving file from git
	git mv 03-ThirdNewFile.py 03-ThirdFile.py

-- cancel changes on staging area
	git restore --staged <file>

-- discard changes on workspace
	git restore <file>

-- remove file from staging area (cancel commit) returns file to workarea still changed
	git reset HEAD <file>
	
-- commit all changes directly without staging
	-- option -a (--All) commits all files
	-- new added files will not be committed. they need to be added to git first.
	commit -am "Commit all changes"
